import { Controller, Get, HttpStatus, Req, UseGuards } from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';
import { AppService } from './app.service';
import { Auth } from './common/decorator/auth.decorator';
import { RequestFBUser } from './common/dto/RequestFBUser.interface';
import { RoleEnum } from './common/enum/role.enum';
import { JwtAuthGuard } from './common/guard/jwt-auth.guard';

@Controller()
export class AppController {
  constructor(private readonly appService: AppService) {}

  @Get()
  getHello(): string {
    return this.appService.getHello();
  }

  @Get("/facebook")
  @UseGuards(AuthGuard("facebook"))
  async facebookLogin(): Promise<any> {
    return HttpStatus.OK;
  }

  @Get("/facebook/redirect")
  @UseGuards(AuthGuard("facebook"))
  async facebookLoginRedirect(@Req() req: RequestFBUser): Promise<any> {
    return {
      statusCode: HttpStatus.OK,
      data: req.user,
    };
  }

  @Get("/testLoginByFacebook")
  @UseGuards(AuthGuard("facebook"))
  async test(@Req() req: RequestFBUser): Promise<any> {
    return {
      statusCode: HttpStatus.OK,
      data: req.user,
    };
  }

  @UseGuards(JwtAuthGuard)
  @Get('profile')
  getProfile(@Req() req) {
    return req.user;
  }

  @Get('/admin')
  @Auth(RoleEnum.ADMIN)
  public testAdmin(@Req() req){
    return req.user;
  }

  @Get("no")
  getNo() {
    return "no";
  }

}
